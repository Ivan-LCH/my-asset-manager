### 2.2 `backend/db/models.py` — SQLAlchemy ORM
```python
from sqlalchemy import Column, String, Float, Integer, Text, ForeignKey
from sqlalchemy.orm import DeclarativeBase, relationship

class Base(DeclarativeBase): pass

class Asset(Base):
    __tablename__ = "assets"
    id = Column(String, primary_key=True)
    type = Column(String, nullable=False)
    name = Column(String, nullable=False)
    current_value = Column(Float, default=0)
    acquisition_date = Column(String)
    acquisition_price = Column(Float, default=0)
    disposal_date = Column(String)
    disposal_price = Column(Float, default=0)
    quantity = Column(Float, default=0)
    created_at = Column(String)
    updated_at = Column(String)

    history = relationship("AssetHistory", back_populates="asset", cascade="all, delete")
    real_estate = relationship("RealEstateDetail", back_populates="asset", uselist=False, cascade="all, delete")
    stock = relationship("StockDetail", back_populates="asset", uselist=False, cascade="all, delete")
    pension = relationship("PensionDetail", back_populates="asset", uselist=False, cascade="all, delete")
    savings = relationship("SavingsDetail", back_populates="asset", uselist=False, cascade="all, delete")

class AssetHistory(Base):
    __tablename__ = "asset_history"
    id = Column(Integer, primary_key=True, autoincrement=True)
    asset_id = Column(String, ForeignKey("assets.id", ondelete="CASCADE"))
    date = Column(String, nullable=False)
    value = Column(Float)
    price = Column(Float)
    quantity = Column(Float)
    asset = relationship("Asset", back_populates="history")

# (RealEstateDetail, StockDetail, PensionDetail, SavingsDetail 동일 패턴)
```

### 2.3 차트 집계 로직 (`backend/db/crud.py`)

핵심 알고리즘: 날짜별 자산 가치 Forward Fill

```python
import pandas as pd
from datetime import datetime, timedelta

def generate_chart_data(assets: list, period: str = "all", group_by: str = "type") -> list:
    """
    assets: get_all_assets()의 결과 (이력 포함)
    period: all|10y|3y|1y|3m|1m
    group_by: type|name|account
    """
    today = datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)
    
    # 기간 계산
    period_days = {"10y": 3650, "3y": 1095, "1y": 365, "3m": 90, "1m": 30}
    if period in period_days:
        start = today - timedelta(days=period_days[period])
    else:
        start = datetime(2015, 1, 1)  # 전체
    
    all_records = []
    for asset in assets:
        # 각 자산의 이력 포인트를 레코드로 변환
        # (취득일 초기값 → 이력 → 현재값/매각값 → 매각 후 0)
        records = _asset_to_records(asset)
        all_records.extend(records)
    
    if not all_records:
        return []
    
    df = pd.DataFrame(all_records)
    df = df.sort_values("date").drop_duplicates(subset=["asset_id", "date"], keep="last")
    df_pivot = df.pivot(index="date", columns="asset_id", values="value")
    
    full_idx = pd.date_range(start=start, end=today, freq="D")
    df_pivot = df_pivot.reindex(full_idx).ffill().fillna(0)
    
    df_melt = df_pivot.reset_index().melt(id_vars="index", var_name="asset_id", value_name="value")
    df_melt.rename(columns={"index": "date"}, inplace=True)
    
    # 메타데이터 JOIN (name, type, account)
    meta_map = {a["id"]: a for a in assets}
    
    # group_by에 따라 집계
    df_melt["date"] = df_melt["date"].dt.strftime("%Y-%m-%d")
    df_melt["label"] = df_melt["asset_id"].map(lambda aid: _get_label(meta_map.get(aid, {}), group_by))
    
    result = df_melt.groupby(["date", "label"])["value"].sum().reset_index()
    return result.rename(columns={"label": "label"}).to_dict(orient="records")
```

const api = axios.create({

// 응답 인터셉터: snake_case → camelCase
api.interceptors.response.use(res => {
  res.data = snakeToCamel(res.data);
  return res;
});
// 요청 인터셉터: camelCase → snake_case

### 3.2 React Query 훅 (`frontend/src/hooks/useAssets.ts`)

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { assetApi } from '@/lib/api';

export function useAssets() {
  return useQuery({
    queryKey: ['assets'],
    queryFn: assetApi.getAll,
    staleTime: 5 * 60 * 1000, // 5분
  });
}

export function useAssetsByType(type: AssetType) {
  const { data: assets } = useAssets();
  return assets?.filter(a => a.type === type) ?? [];
}

export function useCreateAsset() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: assetApi.create,
    onSuccess: () => qc.invalidateQueries({ queryKey: ['assets'] }),
  });
}

export function useDeleteAsset() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) => assetApi.delete(id),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['assets'] }),
  });
}
```

## 4. 특수 기능 구현 주의사항

### 4.1 주식 수량 전파 로직
```
수량 변경 감지 → 해당 날짜 이후 모든 이력 조회 → 수량 덮어쓰기 (단가는 유지) → value 재계산 (price * new_qty * rate)
주의: 미래 전파 시 환율은 동일한 rate 적용 (다중 API 호출 방지)
```

### 4.2 부동산 순자산 계산
```
equity = current_value - loan_amount - tenant_deposit
(부채 = 대출금 + 보증금)
차트에서 value = current_value - (loan + deposit) 로 집계
```

### 4.3 연금 시뮬레이션
```
각 연도별 수령액 = expected_monthly_payout * (1 + annual_growth_rate/100)^(year - start_year)
단, expected_start_year <= year <= expected_end_year 인 경우만
is_pension_like 인 주식/예적금 자산도 포함
```
